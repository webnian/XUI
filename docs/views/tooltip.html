      .tooltip {
        cursor: default;
      }
      .tooltip-box {
        background: #444;
        color: #fff;
        font-size: 14px;
        max-width: 250px;
        padding: 3px 5px;
        position: absolute;
        top: -300px;
        left: -9999px;
        z-index: 2000;
        border-radius: 4px;
        display: none;
        word-wrap: break-word;
      }
		
      var tooltip = ''
      $(document).find('.tooltip').hover(function(e) {
        var el = $(this)
        var title = el.data('title') || ''
        if (!title) return
        if (tooltip) {
          tooltip.text(title).fadeIn(300)
        } else {
          tooltip = $('
'+ title +'
').appendTo('body').fadeIn(300)
        }
        el.mousemove(function(e) {
          tooltip.css({ top: e.pageY + 8, left: e.pageX + 15 })
        })
      }, function() {
        tooltip && tooltip.hide()
      })
